            start == NULL

			    copy from accountno in the record whose address is in temp to tempNextAccNum

        while ( uaccountno != tempNextAccNum && tempNext != NULL)

				copy from next in the record whose address is in tempNextNext to tempNextNext

			while ( tempNextAccNum == accountno && tempNext != NULL)
				copy from next in the record whose address is in tempNext to tempNext
