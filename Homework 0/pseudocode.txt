--------------------------------------------------------------------------------
*** Assume the following variables are already defined. ***

- numbers: array of int
- n: the size of the integer array (numbers)
- m: the number of first slots which already have values (m < n)
- num: the number to insert to the array
--------------------------------------------------------------------------------

define an int called loopTracker
copy from 1 to loopTacker

define an int called innerLoopTracker
define an int called loopCounter
define an int called breakChecker

if(m = 0)
    copy from num to numbers[m]
    copy from m + 1 to m
else 
    copy from m + 1 to m
    copy from num to numbers[m]
    while(loopTracker == 1)
        copy from 1 to innerLoopTracker
        copy from 0 to loopCounter
        copy from 0 to breakChecker

        while(innerLoopTracker == 1)
            if(numbers[0 + loopCounter] > numbers[1 + loopCounter])
                copy from 1 to breakChecker
                copy from numbers[0 + loopCounter] to numbers[m]
                copy from numbers[1 + loopCounter] to numbers[0 + loopCounter]
                copy from numbers[m] to numbers[1 + loopCounter]
            
            copy from loopCounter + 1 to loopCounter

            if(loopCounter == (m - 1))
                copy from 0 to innerLoopTracker
        
        if(breakChecker == 0)
            copy from 0 to loopTacker
